#include "pch.h"
#include <iostream>

using namespace std;

int main()
{
	setlocale(LC_ALL, "RUS");

	int n = 0;
	int k = 0;
	int i = 0;
	int amount = 0;
	int max = -1;
	int count = 0;


	cout << "¬ведите N, K: ";
	cin >> n >> k;
	cout << endl;

	int *arr = new int[n + 1];

	for (i = 0; i <= n+1; i++) {
		arr[i] = 0;
	}
	arr[0] = 1;
	arr[n + 1] = 1;
	max = n;

	for (count = 1; count <= k; count++) {
		max /= 2;
		arr[max] = 1;
		for (i = 1; i <= n; i++) {
			if (arr[i] == 0) {
				amount++;
				if (amount > max) {
					max = amount;
				}
				else {
					amount = 0;
				}
			}
		}
		


	}

	for (i = 0; i <= n + 1; i++) {
		cout << arr[i];
	}
}
